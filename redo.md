```
import numpy as np
from sklearn.metrics.pairwise import cosine_similarity

# –¥–æ–ø—É—Å—Ç–∏–º, —Ç—ã —É–∂–µ –∑–∞–≥—Ä—É–∑–∏–ª–∞ –¥–∞–Ω–Ω—ã–µ –∏–∑ –±–∞–∑—ã
averaged_vectors = np.load("try/vector_db_vectors.npy")  # (N, 512)
cluster_metadata = np.load("try/vector_db_meta.npy", allow_pickle=True).tolist()

def find_similar_face(query_vec, threshold=0.6):
    """
    –ù–∞—Ö–æ–¥–∏—Ç –±–ª–∏–∂–∞–π—à–∏–π —É—Å—Ä–µ–¥–Ω—ë–Ω–Ω—ã–π –≤–µ–∫—Ç–æ—Ä –≤ –±–∞–∑–µ.
    –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –º–µ—Ç–∞–¥–∞–Ω–Ω—ã–µ –∫–ª–∞—Å—Ç–µ—Ä–∞, –µ—Å–ª–∏ —Å—Ö–æ–∂–µ—Å—Ç—å –≤—ã—à–µ –ø–æ—Ä–æ–≥–∞.
    """
    # –Ω–æ—Ä–º–∞–ª–∏–∑—É–µ–º –≤–µ–∫—Ç–æ—Ä
    query_vec = query_vec / np.linalg.norm(query_vec)

    # –∫–æ—Å–∏–Ω—É—Å–Ω—ã–µ —Å—Ö–æ–¥—Å—Ç–≤–∞
    similarities = cosine_similarity([query_vec], averaged_vectors)[0]
    best_idx = np.argmax(similarities)
    best_score = similarities[best_idx]

    if best_score >= threshold:
        print(f"‚úÖ –ù–∞–π–¥–µ–Ω–æ —Å–æ–≤–ø–∞–¥–µ–Ω–∏–µ: {best_score:.3f}")
        return cluster_metadata[best_idx]
    else:
        print(f"‚ö†Ô∏è –°–æ–≤–ø–∞–¥–µ–Ω–∏–π –Ω–µ –Ω–∞–π–¥–µ–Ω–æ (–º–∞–∫—Å. —Å—Ö–æ–∂–µ—Å—Ç—å {best_score:.3f})")
        return None
```
